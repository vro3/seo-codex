This guide demonstrates three methods to load show data into the SEO Codex system.

Method 1: Load from GitHub/Remote URL

Fetch data from any web-accessible JSON file (GitHub, Google Sheets, hosted file).



// Example code for remote loading
window.SHOW_DATA_URL = "YOUR_URL_HERE";

async function loadShows() {
  const response = await fetch(window.SHOW_DATA_URL);
  if (!response.ok) throw new Error('Failed to fetch');
  const data = await response.json();
  return data;
}

Method 2: Load from Local File

Upload a JSON file from your computer directly into the browser.


// Example code for local file loading
function loadLocalFile(file) {
  const reader = new FileReader();
  reader.onload = (e) => {
    const data = JSON.parse(e.target.result);
    window.SHOW_DATA = data;
    console.log('Loaded', data.shows.length, 'shows');
  };
  reader.readAsText(file);
}

Method 3: Pre-load Data in Script

Embed data directly in your HTML (useful for testing or small datasets).

// Embed data directly
window.SHOW_DATA = {
  "shows": [
    {
      "title": "AI DJ Experience",
      "meta_title": "AI DJ - Interactive Music",
      "meta_description": "Experience the future...",
      "page_slug": "ai-dj-experience",
      "tags": ["ai", "dj", "interactive"],
      "imageurl": "https://example.com/img.jpg"
    }
  ]
};
